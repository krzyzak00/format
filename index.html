Tiny jQuery string format plugin

Adds Python-like format function to jQuery

Usage examples:

$.fn.format('The quick {} {} jumped over the {} {}', "brown", "fox", "lazy", "dog");

$.fn.format('The quick {0} {2} jumped over the {1} {3}', "brown", "lazy", "fox", "dog");

$.fn.format('The quick {_brown_} {_fox_} jumped over the {_lazy_} {_dog_}', { _brown_: "brown", _fox_: "Fox", _lazy_: "lazy", _dog_: "Dog" });

$.fn.format('The quick {0} {1} jumped over the {2} {3}', { 0: "brown", 1: "fox", 2: "lazy", 3: "dog" });
